# Commands to run after creating your GitHub repository
# Replace YOUR_USERNAME with your actual GitHub username

# 1. Add the remote repository
git remote add origin https://github.com/YOUR_USERNAME/realtime-api-backend.git

# 2. Rename the default branch to main (modern convention)
git branch -M main

# 3. Push your code to GitHub
git push -u origin main

# After running these commands, your code will be live on GitHub!
# Your repository URL will be: https://github.com/YOUR_USERNAME/realtime-api-backend
